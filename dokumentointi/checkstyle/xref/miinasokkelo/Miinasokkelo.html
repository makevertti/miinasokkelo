4   import javax.swing.JFrame;
5   import javax.swing.JOptionPane;
6   import miinasokkelo.logiikka.Pelialue;
7   import miinasokkelo.logiikka.Pelaaja;
8   
9   /**
10   * Main metodin sisältävä luokka, joka käynnistää pelin 
11   */
12  public class Miinasokkelo {
13      private static int pelialueenKoko = 30;
14      private static int miinat = 80;
15      private static String[] vaikeustasot = {"Helppo", "Normaali", "Vaikea"};
16      private static String vaikeustaso;
17      
18      public static void main(String[] args) {
19          vaikeustaso = (String) JOptionPane.showInputDialog(new JFrame(), "Valitse vaikeustaso", "miinasokkelo", JOptionPane.QUESTION_MESSAGE, null, vaikeustasot, vaikeustasot[0]);
20          if (vaikeustaso == null) {
21              System.exit(0);
22          } else if (vaikeustaso.equals("Helppo")) {
23              pelialueenKoko = 20;
24              miinat = 32;
25          } else if (vaikeustaso.equals("Normaali")) {
26              pelialueenKoko = 25;
27              miinat = 75;
28          } else if (vaikeustaso.equals("Vaikea")) {
29              pelialueenKoko = 30;
30              miinat = 100;
31          }
32          Pelialue pelialue = new Pelialue(pelialueenKoko, miinat, new Pelaaja(0, 0), true);
33          pelialue.setNappaimistonkuuntelija();
34      }
35  }
