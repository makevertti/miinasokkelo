2   package miinasokkelo.kayttoliittyma;
3   
4   import java.awt.Color;
5   import javax.swing.JLabel;
6   import javax.swing.JPanel;
7   
8   /**
9    * Graafisen pelialueen jokainen ruutu on tämän luokan ilmentymä
10   */
11  public class GrafiikkaRuutu extends JPanel {
12      
13      /**
14       * Vaihtaa ruudun tyypin
15       * 
16       * @param tyyppi Ruudun tyyppi. Vaihtoehdot: 
17       *               "tuntematon", "tyhjä", "miina", 
18       *               "pelaaja", "törmätty miina", "maali"
19       */
20      public void vaihdaRuudunTyyppi(String tyyppi) {
21          switch (tyyppi) {
22              case "tuntematon":
23                  this.setBackground(Color.LIGHT_GRAY);
24                  break;
25  
26              case "tyhjä":
27                  this.setBackground(Color.WHITE);
28                  break;
29  
30              case "miina":
31                  this.setBackground(Color.PINK);
32                  break;
33  
34              case "pelaaja":
35                  this.setBackground(Color.CYAN);
36                  break;
37  
38              case "törmätty miina":
39                  this.setBackground(Color.RED);
40                  break;
41  
42              case "maali":
43                  this.setBackground(Color.GREEN);
44                  break;
45          }
46      }
47  
48      /**
49       * Lisää ruutuun numeron, mikä kertoo kuinka
50       * monta miinaa ruudun ympärillä on
51       * 
52       * @param miinatRuudunYmparilla     Lisättävä numero
53       */
54      public void lisaaMiinaNumero(int miinatRuudunYmparilla) {
55          if (this.getComponentCount() < 1 && miinatRuudunYmparilla > 0) {
56              this.add(new JLabel(miinatRuudunYmparilla + ""));
57              this.revalidate();
58              this.repaint();
59          }
60      }
61  }
